# 02_演算子オーバーロード
```cpp
class Vector2i {
public:
    Vector2i() {}
    Vector2i(int x, int y) : 
        x(x),
        y(y)
    {
    }
    int x;
    int y;
    Vector2i operator+ (const Vector2i &o) const {
        return Vector2i(x + o.x, y + o.y);
    }
    Vector2i operator- (const Vector2i &o) const {
        return Vector2i(x - o.x, y - o.y);
    }
};
```

# 03_テンプレート
```cpp
template<class T>
class Vector2 {
public:
    Vector2(T x, T y) : 
        x(x),
        y(y)
    {}
    T x;
    T y;

    Vector2 operator+(const Vector2 &o) const {
        return Vector2(x + o.x, y + o.y);
    }
    Vector2 operator-(const Vector2 &o) const {
        return Vector2(x - o.x, y - o.y);
    }
};
```

# 04_コピーコンストラクタと代入演算子
```cpp
class MyClass {
public:
    MyClass(int x, int y) : 
        x(x),
        y(y)
    {
    }
    MyClass(const MyClass &o) : 
        x(o.x),
        y(o.y)
    {
    }
    MyClass &operator= (const MyClass &) {
        x = o.x;
        y = o.y;
    }
    int x;
    int y;
};
```

# 05_unique_ptr
```cpp
#include<iostream>

    std::cout << (*p) << std::endl;
}

	std::unique_ptr<int> p = std::make_unique<int>(10);
	f(p.get());
	return 0;
}
```
