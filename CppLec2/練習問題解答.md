# 02_演算子オーバーロード
```cpp
class Vector2i {
public:
    Vector2i() {}
    Vector2i(int x, int y) : 
        x(x),
        y(y)
    {
    }
    int x;
    int y;
    Vector2i operator+ (const Vector2i &o) const {
        return Vector2i(x + o.x, y + o.y);
    }
    Vector2i operator- (const Vector2i &o) const {
        return Vector2i(x - o.x, y - o.y);
    }
};
```

```cpp
template<class T>
class Vector2 {
public:
    Vector2(T x, T y) : 
        x(x),
        y(y)
    {}
    T x;
    T y;

    Vector2 operator+(const Vector2 &o) const {
        return Vector2(x + o.x, y + o.y);
    }
    Vector2 operator-(const Vector2 &o) const {
        return Vector2(x - o.x, y - o.y);
    }
};
```

# 06_list
```cpp
#include<list>
#include<iostream>

int main() {
    std::list<int> ls;
    for(int i = 0; i < 10; i++) {
        ls.emplace_back(i);
    }
    for(const auto &i : ls) {
        std::cout << i << std::endl;
    }
    auto itr = ls.begin();
    itr++;
    ls.erase(itr);
    for(const auto &i : ls) {
        std::cout << i << std::endl;
    }
    return 0;
}
```

# 08_unique_ptr
```cpp
#include<iostream>
#include<memory>

int *f(int *p) {
	return std::make_unique<int>(20).release();
}

int main() {
	std::unique_ptr<int> p = std::make_unique<int>(10);
	std::cout << (*p) << std::endl;
	int *tp = f(p.get());

	p.reset(tp);

	std::cout << (*p) << std::endl;

	return 0;
}
```
